     { List<Ponto> pts = super.getPoligono();
       Ponto p1 = pts.get(0);
       Ponto p2 = pts.get(1);
       double lado1 = p1.distancia(p2); 
       Ponto p3 = pts.get(2);
       double lado2 = p2.distancia(p3);
